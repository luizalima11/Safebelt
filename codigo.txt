$sql = "INSERT INTO usuarios (usuario, senha) VALUES (?, ?)";
$stmt = $conexao->prepare($sql);
$stmt->bind_param("ss", $usuario, $senha);

if ($stmt->execute()) {
    header("Location: login.html");
    exit();
} else {

    if ($stmt->errno == 1062) {

login.php ==============================================================
<?php
include "conexao.php";

$usuario = $_POST['usuario'];
$senha = md5($_POST['senha']);

$sql = "SELECT * FROM usuarios WHERE usuario = ?";
$stmt = $conexao->prepare($sql);
$stmt->bind_param("s", $usuario);
$stmt->execute();
$result = $stmt->get_result();

if ($result->num_rows == 1) {
    
    $dados = $result->fetch_assoc();

    if ($dados['senha'] === $senha) {
        header("Location: inicio.html");
        exit();
    } else {
        echo "Senha incorreta!";
    }

} else {
    echo "Usuário não encontrado!";
}
?>

    async function atualizarDados() {
      try {
        const resposta = await fetch("dados.php");
        const dados = await resposta.json();

        if (!dados || dados.erro) return;

        document.getElementById("batimento").textContent = `${dados.batimento} bpm`;
        document.getElementById("giroscopio").textContent = `X: ${dados.giro_x} | Y: ${dados.giro_y} | Z: ${dados.giro_z}`;

        document.getElementById("gps").textContent = `Lat: ${dados.latitude} | Lon: ${dados.longitude}`;

        const alerta = document.getElementById("alerta");
        const status = document.getElementById("status-crise");
        const statusTexto = document.getElementById("status-texto");

        if (dados.batimento > 110) {
          alerta.style.display = "block";
          status.className = "status-crise status-alerta";
          statusTexto.textContent = "⚠️ Sinais de possível crise detectados!";
        } else {
          alerta.style.display = "none";
          status.className = "status-crise status-normal";
          statusTexto.textContent = "💙 Estudante estável";
        }

      } catch (e) {
        console.error("Erro ao buscar dados:", e);
      }
    }

    atualizarDados();
    setInterval(atualizarDados, 5000);
